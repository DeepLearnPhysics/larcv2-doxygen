388  }else{

392 
393  // Make sure max entry does not exceed the physical max from input. If so, truncate.
394  if(_io.io_mode() != IOManager::kWRITE && max_entry > _access_entry_v.size()){
395  LARCV_WARNING() << "Requested to process entries from " << start_entry << " to " << max_entry-1
396  << " ... but there are only " << _access_entry_v.size() << " entries in input!" << std::endl
397  << "Truncating the end entry to " << _access_entry_v.size()-1 << std::endl;
398  max_entry = _access_entry_v.size();
399  }
400 
401  // Batch-execute in while loop
402  size_t num_processed=0;
403  size_t num_fraction=(max_entry - _current_entry)/10;
404  while(_current_entry < max_entry) {

406  if(_io.io_mode() != IOManager::kWRITE) {
407  //_io.clear_entry();

413  if(!num_fraction) {
