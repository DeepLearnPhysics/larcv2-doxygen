168  std::vector<float> Image2D::copy_compress(size_t rows, size_t cols, CompressionModes_t mode) const
169  {
170  if (mode == kOverWrite)
171  throw larbys("kOverWrite is invalid for copy_compress!");
172 
173  const size_t self_cols = _meta.cols();
174  const size_t self_rows = _meta.rows();
175  if (self_cols % cols || self_rows % rows) {
176  char oops[500];
177  sprintf(oops, "Compression only possible if height/width are modular 0 of compression factor! H:%zuMOD%zu=%zu W:%zuMOD%zu=%zu",
178  self_rows, rows, self_rows % rows, self_cols, cols, self_cols % cols);
179  throw larbys(oops);
180  }
181  size_t cols_factor = self_cols / cols;
182  size_t rows_factor = self_rows / rows;
183  std::vector<float> result(cols * rows, 0);
184 
185  for (size_t col = 0; col < cols; ++col) {
186  for (size_t row = 0; row < rows; ++row) {
187  float value = 0;
188 
189  for (size_t orig_col = col * cols_factor; orig_col < (col + 1)*cols_factor; ++orig_col)
190  for (size_t orig_row = row * rows_factor; orig_row < (row + 1)*rows_factor; ++orig_row) {
191 
192  if ( mode != kMaxPool ) {
193  // for sum and average mode
194  value += _img[orig_col * self_rows + orig_row];
195  }
196  else {
197  // maxpool
198  value = ( value < _img[orig_col * self_rows + orig_row] ) ? _img[orig_col * self_rows + orig_row] : value;
199  }
200  }
201 
202  result[col * rows + row] = value;
203  if ( mode == kAverage )
204  result[col * rows + row] /= (float)(rows_factor * cols_factor);
205  }
206  }
207  return result;
208  }

315  double y_max = std::min(_meta.max_y(), rhs_meta.max_y());

319  size_t col_min1 = _meta.col(x_min);
320

322  size_t col_min2 = rhs_meta.col(x_min);
323

325  size_t ncols = (x_max - x_min) / _meta.pixel_width();

328

330

333
